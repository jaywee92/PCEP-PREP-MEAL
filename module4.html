Functions & Exceptions

Funktionsdefinition

def function_name(parameter1, parameter2):
    return result

Parameter Typen

# Positional
def func(a, b):
    pass

# Default
def func(a, b=10):
    pass

# *args
def func(*args):
    for arg in args:
        print(arg)

# **kwargs
def func(**kwargs):
    for key, val in kwargs.items():
        print(key, val)

Return Statement

def add(a, b):
    return a + b

def get_coords():
    return 10, 20

x, y = get_coords()

Lambda-Funktionen

square = lambda x: x ** 2
add = lambda x, y: x + y

even = list(filter(lambda x: x % 2 == 0, [1,2,3,4]))

Variable Scope

global_var = 10

def func():
    local_var = 20
    print(global_var)

def modify_global():
    global global_var
    global_var = 20

Exception Handling

try:
    num = int("abc")
except ValueError:
    print("Invalid!")
except TypeError:
    print("Wrong type!")
else:
    print(f"OK: {num}")
finally:
    print("Always runs")

raise Statement

def check_age(age):
    if age < 0:
        raise ValueError("Age cannot be negative!")
    return age

Common Exceptions

ValueError  # Invalid value
TypeError  # Wrong type
KeyError  # Key missing
IndexError  # Index out of range
ZeroDivisionError  # Division by zero
